n=int(input())
l=[]
for i in range(0,n,1):
    x=int(input())
    l.append(x)
for i in range(0,n-1,1):
    for j in range(i+1,n,1):
        if l[i]<l[j]:
            t=l[i]
            l[i]=l[j]
            l[j]=t
for i in range(0,n,1):
    print(l[i],end=" ")
